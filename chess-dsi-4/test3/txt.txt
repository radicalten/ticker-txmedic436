int history_count = 0;
Step 2: Cache SAN Notation on Move Submission
Update the push_state() function to compute the SAN string immediately when a move is recorded:
void push_state(const BoardState *state, Move m) {
    if (history_count < MAX_HISTORY - 1) {
        history[history_count] = *state;
        move_history[history_count] = m;
        
        // Compute and cache the SAN string once right here
        move_to_san(state, m, san_history[history_count]); 
        
        history_count++;
    }
}
